######问题描述
    如果a是b的朋友，b是c的朋友，则a是c的朋友，a是b的朋友等价于b是a的朋友
    输入的第一行有两个数 n 代表人数，m代表朋友组数，接下来有m行，每行有两个正整数a和b，表示a和b是好朋友
    输出一个整数，表示可以分成几组
```
#include<cstdio>
#include<cstring>
using namespace std;
const int N = 105;
const int M = 105;
int father[N];
int findFather(int v)
{
    if (v == father[v])
        return v;
    else
    {
        int f = findFather(father[v]);
        father[v] = f;
        return f;
    }
}
void merge(int a, int b)
{
    int fa = findFather(a);
    int fb = findFather(b);
    if (fa != fb)
        father[fa] = fb;
}
void init(int n)
{
    for (int i = 1; i <= n; i++)
    {
        father[i] = i;
    }
}
int main()
{
    int n, m, a, b;
    scanf("%d%d", &n, &m);
    for (int i = 0; i <= N; i++)
    {
        father[i] = i;//令father[i]为-1也可，此处以father[i]=i为例
    }
    init(n);
    for (int i = 0; i < m; i++)
    {
        scanf("%d%d", &a, &b);
        merge(a, b);
    }
    int num = 0;
    for (int i = 1; i <= n; i++)
    {
        if (father[i] == i)
            num++;
    }
    printf("%d\n", num);
    return 0;
}
样例一
4 2
1 4
2 3
输出 2

样例二
7 5
1 2
2 3
3 1
1 4
5 6
输出 5
```
